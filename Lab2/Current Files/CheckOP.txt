module CheckOP(
    input OP0,
    input OP1,
    input [15:0] A,
    input [15:0] B,
    output [15:0] OUTA,
    output [15:0] OUTB
    );
    wire wOP;

    and AND1oa (wOP, OP0, OP1);
    
    //if the above is true, the values of A and B will pass on
    //otherwise, the values will be set to 0
    and
        AND2oa (OUTA, A, wOP),
        AND3oa (OUTB, B, wOP);
         
endmodule
